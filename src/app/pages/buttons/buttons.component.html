基礎樣式

// 預設按鈕 (淺灰背景)
<button class="ft-btn">Default</button>

彩色按鈕

<button class="ft-btn ft-btn-primary">主要按鈕</button>
<button class="ft-btn ft-btn-secondary">次要按鈕</button>
<button class="ft-btn ft-btn-accent">強調按鈕</button>
<button class="ft-btn ft-btn-success">成功按鈕</button>
<button class="ft-btn ft-btn-danger">危險按鈕</button>
<button class="ft-btn ft-btn-warning">警告按鈕</button>

玻璃質感 (深色背景適用)

<button class="ft-btn ft-btn-glass">玻璃質感主要按鈕</button>
<button class="ft-btn ft-btn-glass-secondary">玻璃質感次要按鈕</button>
<button class="ft-btn ft-btn-glass-clear">全透明玻璃質感按鈕</button>

玻璃質感 (淺色背景適用)

<button class="ft-btn ft-btn-glass-light">淺色玻璃質感主要按鈕</button>
<button class="ft-btn ft-btn-glass-light-secondary">淺色玻璃質感次要按鈕</button>
<button class="ft-btn ft-btn-glass-dark">暗色玻璃質感按鈕</button>

尺寸修飾符

<button class="ft-btn ft-btn-sm ft-btn-primary">小尺寸按鈕</button>
<button class="ft-btn ft-btn-primary">預設尺寸按鈕</button>
<button class="ft-btn ft-btn-lg ft-btn-primary">大尺寸按鈕</button>

Icon 按鈕

// 帶文字的 Icon 按鈕
<button class="ft-btn ft-btn-primary">
  <span class="material-icons">add</span>
  新增
</button>

// Icon 在文字後面
<button class="ft-btn ft-btn-icon-end ft-btn-primary">
  下一步
  <span class="material-icons">arrow_forward</span>
</button>

// 純 Icon 按鈕 (正方形)
<button class="ft-btn ft-btn-icon ft-btn-primary">
  <span class="material-icons">add</span>
</button>

// 圓形 Icon 按鈕
<button class="ft-btn ft-btn-icon ft-btn-rounded ft-btn-primary">
  <span class="material-icons">add</span>
</button>

按鈕群組

// 預設靠左對齊
<div class="ft-button-group">
  <button class="ft-btn ft-btn-primary">Button 1</button>
  <button class="ft-btn ft-btn-secondary">Button 2</button>
</div>

// 置中對齊
<div class="ft-button-group justify-center">
  <button class="ft-btn ft-btn-primary">Button 1</button>
  <button class="ft-btn ft-btn-secondary">Button 2</button>
</div>

// 靠右對齊
<div class="ft-button-group justify-end">
  <button class="ft-btn ft-btn-primary">Button 1</button>
  <button class="ft-btn ft-btn-secondary">Button 2</button>
</div>
